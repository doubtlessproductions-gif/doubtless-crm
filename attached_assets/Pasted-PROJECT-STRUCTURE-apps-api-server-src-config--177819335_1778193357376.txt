export const signAccessToken = (payload: any) => {
  return jwt.sign(payload, authConfig.accessSecret, {
    expiresIn: authConfig.accessExpires,
  });
};

export const signRefreshToken = (payload: any) => {
  return jwt.sign(payload, authConfig.refreshSecret, {
    expiresIn: authConfig.refreshExpires,
  });
};
AUTH SERVICE
src/services/auth.service.ts
import bcrypt from "bcrypt";
import { prisma } from "../plugins/prisma";
import {
  signAccessToken,
  signRefreshToken,
} from "../utils/tokens";

export const loginUser = async (
  email: string,
  password: string
) => {
  const user = await prisma.user.findUnique({
    where: { email },
  });

  if (!user) {
    throw new Error("User not found");
  }

  const valid = await bcrypt.compare(
    password,
    user.password
  );

  if (!valid) {
    throw new Error("Invalid credentials");
  }

  const payload = {
    userId: user.id,
    role: user.role,
    tokenVersion: user.tokenVersion,
  };

  const accessToken = signAccessToken(payload);

  const refreshToken = signRefreshToken(payload);

  await prisma.user.update({
    where: { id: user.id },
    data: {
      refreshToken,
    },
  });

  return {
    accessToken,
    refreshToken,
    user,
  };
};
AUTH MIDDLEWARE
src/middleware/auth.ts
import jwt from "jsonwebtoken";
import { FastifyReply, FastifyRequest } from "fastify";

import { authConfig } from "../config/auth";
import { prisma } from "../plugins/prisma";
import { logger } from "../utils/logger";

export const authMiddleware = async (
  req: FastifyRequest,
  reply: FastifyReply
) => {
  try {
    const authHeader =
      req.headers.authorization;

    if (!authHeader) {
      return reply.status(401).send({
        error: "No token",
      });
    }

    const token = authHeader.split(" ")[1];

    const decoded: any = jwt.verify(
      token,
      authConfig.accessSecret
    );

    const user = await prisma.user.findUnique({
      where: {
        id: decoded.userId,
      },
    });

    if (!user) {
      return reply.status(401).send({
        error: "User missing",
      });
    }

    if (
      decoded.tokenVersion !==
      user.tokenVersion
    ) {
      return reply.status(401).send({
        error: "Token expired",
      });
    }

    req.user = user;
  } catch (err: any) {
    logger.warn({
      err: err.message,
      authHeader:
        req.headers.authorization?.slice(0, 25),
      jwtLoaded: !!process.env.JWT_SECRET,
    });

    return reply.status(401).send({
      error: "Invalid token",
    });
  }
};

import { FastifyReply, FastifyRequest } from "fastify";

export const requireRole =
  (roles: string[]) =>
  async (
    req: FastifyRequest,
    reply: FastifyReply
  ) => {
    const user: any = req.user;

    if (!roles.includes(user.role)) {
      return reply.status(403).send({
        error: "Forbidden",
      });
    }
  };
AUTH ROUTES
src/routes/auth.ts
import { FastifyInstance } from "fastify";
import bcrypt from "bcrypt";

import { prisma } from "../plugins/prisma";
import { loginUser } from "../services/auth.service";

export async function authRoutes(
  app: FastifyInstance
) {
  app.post("/register", async (req, reply) => {
    const body: any = req.body;

    const hashed = await bcrypt.hash(
      body.password,
      10
    );

    const user = await prisma.user.create({
      data: {
        email: body.email,
        password: hashed,
      },
    });

    return user;
  });

  app.post("/login", async (req, reply) => {
    const body: any = req.body;

    const data = await loginUser(
      body.email,
      body.password
    );

    return data;
  });

  app.post("/logout", async (req, reply) => {
    return {
      success: true,
    };
  });
}

export const transcodeVideo = (
  input: string,
  output: string
) => {
  return new Promise((resolve, reject) => {
    ffmpeg(input)
      .output(output)
      .videoCodec("libx264")
      .on("end", resolve)
      .on("error", reject)
      .run();
  });
};

MAIN SERVER
src/index.ts
import Fastify from "fastify";
import cors from "@fastify/cors";
import helmet from "@fastify/helmet";
import rateLimit from "@fastify/rate-limit";

import { logger } from "./utils/logger";

import { authRoutes } from "./routes/auth";
import { userRoutes } from "./routes/users";
import { dashboardRoutes } from "./routes/dashboard";
import { paymentRoutes } from "./routes/payments";

const app = Fastify({
  logger: true,
});

async function start() {
  await app.register(cors);

  await app.register(helmet);

  await app.register(rateLimit, {
    max: 100,
    timeWindow: "1 minute",
  });

  app.register(authRoutes, {
    prefix: "/api/auth",
  });

  app.register(userRoutes, {
    prefix: "/api/users",
  });

  app.register(dashboardRoutes, {
    prefix: "/api/dashboard",
  });

  app.register(paymentRoutes, {
    prefix: "/api/payments",
  });

  app.get("/", async () => {
    return {
      status: "Doubtless API Online",
    };
  });

  await app.listen({
    port: 8080,
    host: "0.0.0.0",
  });

  logger.info(
    "Doubtless infrastructure online"
  );
}

start();
